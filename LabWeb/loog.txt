SELECT user.uid uid,user.name name,userdetail.sex sex,userdetail.degree degree,userdetail.grade grade,printcount.papersum papersum,printarrange.paperlimit paperlimit,SUM(printaddition.addnum) addnum FROM lab_user user LEFT JOIN lab_userdetail userdetail ON userdetail.uid = user.uid LEFT JOIN lab_printcount printcount ON printcount.uid = user.uid AND printcount.month = '2014-07-1' LEFT JOIN lab_printarrange printarrange ON printarrange.uid = user.uid LEFT JOIN lab_printaddition printaddition ON printaddition.uid = user.uid AND printaddition.month = '2014-07-1'  WHERE user.uid LIKE '%%' OR user.name LIKE '%%' OR userdetail.grade LIKE '%%' ORDER BY `uid` asc LIMIT 0,50
